237 void QwtLegendLabel::setDown( bool down )
238 {
239  if ( down == m_data->isDown )
240  return;
241 
242  m_data->isDown = down;
243  update();
244 
245  if ( m_data->itemMode == QwtLegendData::Clickable )
246  {
247  if ( m_data->isDown )
248  Q_EMIT pressed();
249  else
250  {
251  Q_EMIT released();
252  Q_EMIT clicked();
253  }
254  }
255 
256  if ( m_data->itemMode == QwtLegendData::Checkable )
257  Q_EMIT checked( m_data->isDown );
258 }

282 void QwtLegendLabel::paintEvent( QPaintEvent* e )
283 {
284  const QRect cr = contentsRect();
285 
286  QPainter painter( this );
287  painter.setClipRegion( e->region() );
288 
289  if ( m_data->isDown )
290  {
291  qDrawWinButton( &painter, 0, 0, width(), height(),
292  palette(), true );
293  }
294 
295  painter.save();
296 
297  if ( m_data->isDown )
298  {
299  const QSize shiftSize = buttonShift( this );
300  painter.translate( shiftSize.width(), shiftSize.height() );
301  }
302 
303  painter.setClipRect( cr );
304 
305  drawContents( &painter );
306 
307  if ( !m_data->icon.isNull() )
308  {
309  QRect iconRect = cr;
310  iconRect.setX( iconRect.x() + margin() );
311  if ( m_data->itemMode != QwtLegendData::ReadOnly )
312  iconRect.setX( iconRect.x() + ButtonFrame );
313 
314  iconRect.setSize( m_data->icon.size() );
315  iconRect.moveCenter( QPoint( iconRect.center().x(), cr.center().y() ) );
316 
317  painter.drawPixmap( iconRect, m_data->icon );
318  }
319 
320  painter.restore();
321 }
